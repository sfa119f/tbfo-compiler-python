S -> IMPORT | DEF | CLASS | MAIN
IMPORT -> FROM A | A | DEF
A -> JADI_IMPORT B
B -> WORD C | WORD IMPORT | WORD DEF
C -> AS IMPORT | AS DEF
JADI_IMPORT -> import
FROM -> JADI_FROM WORD
JADI_FROM -> from
AS -> JADI_AS WORD
JADI_AS -> as
DEF -> DEF_FUNGSI ISI_FUNG | DEF_PROSEDUR ISI_PRO
DEF_FUNGSI -> JADI_DEF OBJECT
OBJECT -> WORD PARA
PARA -> JADI_KB BELAKANG
BELAKANG -> PARAMETER JADI_KTFUNG | PARAMETER JADI_KTPRO
JADI_KTFUNG -> JADI_KT PANAH
JADI_KB -> (
JADI_KT -> )
PANAH -> SETRIP LEBIHBESAR
LEBIHBESAR -> LEBIH_BESAR TIPE_VAR
TIPE_VAR -> TYPE WORD
SETRIP -> TITIKDUA STRIP
STRIP -> -
TITIKDUA -> :
LEBIH_BESAR -> BESAR TIPE_VAR
BESAR -> >
PARAMETER -> TIPE_VAR PARAMETER_LANJUT | TIPE_VAR
PARAMETER_LANJUT -> KOMA PARAMETER
KOMA -> ,
TYPE -> int | char | str | long | float | bool
DEF_PROSEDUR -> JADI_DEF OBJECT
JADI_KTPRO -> JADI_KT TITIKDUA
ISI_FUNG -> ALGORITMA RETURN | ISI_FUNG ISI_FUNG
RETURN -> RETWORD CLASS | RETWORD MAIN
RETWORD -> RET WORD
RET -> return
ISI_PRO -> ALGORITMA CLASS | ALGORITMA ALGORITMA | ALGORITMA MAIN
CLASS -> KELAS KELAS_LANJ
KELAS -> class
KELAS_LANJ -> NAMAKELAS ISI_KELAS
NAMAKELAS -> WORD TITIKDUA
ISI_KELAS -> DEF CLASS | DEF MAIN | DEF DEF
DEF -> def 
MAIN -> ALGORITMA ALGORITMA | ALGORITMA
