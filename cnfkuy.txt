S -> IMPORT | DEF | CLASS | MAIN
IMPORT -> FROM A | A | DEF
A -> JADI_IMPORT B
B -> KATA C | KATA IMPORT | KATA DEF
C -> AS IMPORT | AS DEF
JADI_IMPORT -> import
FROM -> JADI_FROM KATA
JADI_FROM -> from
AS -> JADI_AS KATA
JADI_AS -> as
DEF -> DEF_FUNGSI ISI_FUNG | DEF_PROSEDUR ISI_PRO | DEF_FUNGSI PASS | DEF_PROSEDUR PASS
PASS -> pass
DEF_FUNGSI -> JADI_DEF OBJECT
OBJECT -> KATA PARA | JADI_KB LANJUTAN
PARA -> JADI_KB BELAKANG 
BELAKANG -> PARAMETER JADI_KTFUNG | PARAMETER JADI_KTPRO
JADI_KTFUNG -> JADI_KT PANAH
JADI_KB -> (
JADI_KT -> )
PANAH -> SETRIP LEBIHBESAR
LEBIHBESAR -> LEBIH_BESAR TIPE_VAR
TIPE_VAR -> TYPE KATA
SETRIP -> TITIKDUA STRIP
STRIP -> -
TITIKDUA -> :
LEBIH_BESAR -> BESAR TIPE_VAR
BESAR -> >
PARAMETER -> TIPE_VAR PARAMETER_LANJUT | TIPE_VAR
PARAMETER_LANJUT -> KOMA PARAMETER
KOMA -> ,
TYPE -> int | char | str | long | float | bool | word
DEF_PROSEDUR -> JADI_DEF OBJECT
JADI_KTPRO -> JADI_KT TITIKDUA 
ISI_FUNG -> ALGORITMA RETURN | ISI_FUNG ISI_FUNG
RETURN -> RETKATA CLASS | RETKATA MAIN
RETKATA -> RET KATA
RET -> return
ISI_PRO -> ALGORITMA CLASS | ALGORITMA ALGORITMA | ALGORITMA MAIN
CLASS -> KELAS KELAS_LANJ
KELAS -> class
KELAS_LANJ -> NAMAKELAS ISI_KELAS
NAMAKELAS -> KATA TITIKDUA
ISI_KELAS -> DEF CLASS | DEF MAIN | DEF DEF
JADI_DEF -> def
DEFCLASS -> JADI_DEF OBJECT
MAIN -> ALGORITMA ALGORITMA | ALGORITMA
ALGORITMA -> LOGIC_STATEMENT | LOOP_STATEMENT | NORMAL_STATEMENT | WITH_STATEMENT
LOGIC_STATEMENT -> IF_STATEMENT ALGORITMA_IF | IF_ELIF_STATEMENT ALGORITMA_IF | IF_ELSE_STATEMENT ALGORITMA_IF
ALGORITMA_IF -> MAIN RAISE
RAISE -> JADI_RAISE WORD
JADI_RAISE -> raise
IF_STATEMENT -> JADI_IF OBJECT
LANJUTAN -> KONDISI JADI_KTPRO
JADI_IF -> if
IF_ELIF_STATEMENT -> IF_STATEMENT ELIF_STATEMENT | IF_STATEMENT NEXT
NEXT -> ELIF_STATEMENT ELSE_STATEMENT
ELIF_STATEMENT -> JADI_ELIF OBJECT
JADI_ELIF -> elif
ELSE_STATEMENT -> JADI_ELSE TITIKDUA
JADI_ELSE -> else 
IF_ELSE_STATEMENT -> IF_STATEMENT ELSE_STATEMENT
KONDISI -> OBJECT | KATA ARI | NUM ARI | KATA LOG | LOG_VAR LOG | STRING LOG_OP 
ARI -> ARI_OP KATA | ARI_OP NUM
LOG -> LOG_OP LOG_VAR | LOG_OP KATA | LOG_OP STRING
STRING -> PETIK2 SUFIKS | PETIK1
SUFIKS -> KEYBOARD PETIK2 | KEYBOARD PETIK1
PETIK2 -> "
PETIK1 -> '
LOG_VAR -> true | false | none
ARI_OP -> KECIL | BESAR | SAMADENGAN SAMADENGAN | TIDAK SAMADENGAN | KECIL SAMADENGAN | BESAR SAMADENGAN
KECIL -> < 
SAMADENGAN -> =
TIDAK -> !
LOG_OP -> SAMADENGAN SAMADENGAN | TIDAK SAMADENGAN | and | or | is
TITIK -> .
KEYBOARD -> KATA | NUM | SYMBOL | KEYBOARD KEYBOARD
SYMBOL -> ! | @ | # | $ | % | ^ | & | * | ( | ) | - | _ | + | = | { | } | [ | ] | \ | : | ; | ' | " | < | > | , | . | / | ? 
LOOP_STATEMENT -> FOR_STATEMENT ALGORITMA_LOOP | WHILE_STATEMENT ALGORITMA_LOOP
FOR_STATEMENT -> FOR BELAKANG_FOR
FOR -> for
BELAKANG_FOR -> KATA IN_FOR
IN_FOR -> IN OBJECT
IN -> in
WHILE_STATEMENT -> WHILE OBJECT
WHILE -> while
ALGORITMA_LOOP -> MAIN BREAK
BREAK -> break | continue | BREAK MAIN
NORMAL_STATEMENT -> KATA DISAMAIN | PRINT KURUNG | PRINT KRG 
DISAMAIN -> SAMADENGAN NUM | SAMADENGAN LOG_VAR | SAMADENGAN STRING | SAMADENGAN INPUT
INPUT -> INP KURUNG
INP -> input
KURUNG -> JADI_KB SETRING | JADI_KB JADI_KT
SETRING -> STRING JADI_KT
PRINT -> print
KRG -> JADI_KB OBJ
OBJ -> STRING JADI_KT | KATA JADI_KT | OBJ KOMAOBJ
KOMAOBJ -> KOMA OBJ
WITH_STATEMENT -> JADI_WITH BELAKANG_WITH
JADI_WITH -> with
BELAKANG_WITH -> JADI_OPEN BELAKANG_OPEN
JADI_OPEN -> open
BELAKANG_OPEN -> JADI_KB BEL_OPEN
BEL_OP -> STRING KOMASTRING | STRING STRINGAKHIR
KOMASTRING -> KOMA STRING
STRINGAKHIR -> JADI_KT BAG_BELAKANGWITH
BAG_BELAKANGWITH -> JADI_AS BELAKANG_AS
BELAKANG_AS -> KATA TITIKDUA
